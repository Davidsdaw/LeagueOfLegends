CREATE TABLE USUARIOS (
    user VARCHAR(50) PRIMARY KEY,
    password VARCHAR(50) NOT NULL,
    rol CHAR(1) CHECK (rol IN ('A', 'R')), -- 'A' para administradores, 'R' para registrados
    mail VARCHAR(100) NOT NULL
);
